
        int main()
 
{
 
  int year;
 
  printf("Please enter a year to check whether it is a leap year or not");
 
  scanf("%d", year);
 
 
 
  if ( year%400 == 0)
 
    printf("%d is a leap year", year);
 
  if ( year%100 == 0)
 
    printf("%d is not a leap year", year);
 
  if ( year%4 == 0 )
 
    printf("%d is a leap year", year);
 
  else
 
    printf("%d is not a leap year", year);  
 
  return 0;
 
}
